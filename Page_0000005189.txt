    BEGIN
      CASE "Wizard Step" OF
        "Wizard Step"::"1":
          BEGIN

            BackEnable := FALSE;
            Step1Visible := Visible;
            IF Visible THEN;
          END;
        "Wizard Step"::"2":
          BEGIN
            Step2Visible := Visible;
            BackEnable := TRUE;
            NextEnable := TRUE;
          END;
        "Wizard Step"::"3":
          BEGIN
            Step3Visible := Visible;
            IF Visible THEN BEGIN

              NextEnable := TRUE;

          END;

          BEGIN
