    BEGIN
      WITH RtngVersion DO BEGIN
        RtngVersion := Rec;
        RoutingHeader.GET("Routing No.");
        RoutingHeader.TESTFIELD("Version Nos.");

          IF STRLEN(VersionCode) > MAXSTRLEN("Version Code") THEN

          Rec := RtngVersion;
          EXIT(TRUE);
        END;
      END;
    END;
