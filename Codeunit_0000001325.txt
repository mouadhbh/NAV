{
  OBJECT-PROPERTIES
  {
    Date=23/09/13;
    Time=12:00:00;
    Version List=NAVW17.10;
  }
  PROPERTIES
  {
    TableNo=122;
    Permissions=TableData 122=rm,
                TableData 124=rm;
    OnRun=BEGIN
            CancelInvoice(Rec);
          END;

  }
  CODE
  {
    VAR
      CancelPostedInvoiceQst@1000 : TextConst 'ENU=The posted purchase invoice will be canceled, and a purchase credit memo will be created and posted, which reverses the posted purchase invoice.\ \Do you want to continue?;FRA=La facture achat enregistr‚e sera annul‚e et un avoir achat sera cr‚‚ et enregistr‚, lequel annule la facture achat enregistr‚e.\ \Souhaitez-vous continuerÿ?';
      OpenPostedCreditMemoQst@1001 : TextConst 'ENU=A credit memo was successfully created. Do you want to open the posted credit memo?;FRA=Un avoir a ‚t‚ correctement cr‚‚. Voulez-vous ouvrir l''avoir enregistr‚ÿ?';

    PROCEDURE CancelInvoice@1(VAR PurchInvHeader@1000 : Record 122) : Boolean;
    VAR
      PurchCrMemoHdr@1002 : Record 124;
      CorrectPostedPurchInvoice@1001 : Codeunit 1313;
    BEGIN
      CorrectPostedPurchInvoice.TestCorrectInvoiceIsAllowed(PurchInvHeader,TRUE);
      IF CONFIRM(CancelPostedInvoiceQst) THEN
        IF CorrectPostedPurchInvoice.CancelPostedInvoice(PurchInvHeader) THEN
          IF CONFIRM(OpenPostedCreditMemoQst) THEN BEGIN
            PurchCrMemoHdr.GET(PurchInvHeader."Canceled By");
            PAGE.RUN(PAGE::"Mini Posted Purch. Credit Memo",PurchCrMemoHdr);
            EXIT(TRUE);
          END;

      EXIT(FALSE);
    END;

    BEGIN
    END.
  }
}
