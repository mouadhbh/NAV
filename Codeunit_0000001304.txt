OBJECT Codeunit 1304 Sales-Quote to Invoice Yes/No
{
  OBJECT-PROPERTIES
  {
    Date=23/09/13;
    Time=12:00:00;
    Version List=NAVW17.10;
  }
  PROPERTIES
  {
    TableNo=36;
    OnRun=VAR
            InvoiceSalesHeader@1001 : Record 36;
            SalesQuoteToInvoice@1000 : Codeunit 1305;
          BEGIN
            TESTFIELD("Document Type","Document Type"::Quote);
            IF GUIALLOWED THEN
              IF NOT CONFIRM(ConfirmConvertToInvoiceQst,FALSE) THEN
                EXIT;

            SalesQuoteToInvoice.RUN(Rec);
            SalesQuoteToInvoice.GetSalesInvoiceHeader(InvoiceSalesHeader);

            COMMIT;

            IF GUIALLOWED THEN
              IF CONFIRM(STRSUBSTNO(OpenNewInvoiceQst,InvoiceSalesHeader."No."),TRUE) THEN
                PAGE.RUN(PAGE::"Mini Sales Invoice",InvoiceSalesHeader);
          END;

  }
  CODE
  {
    VAR
      ConfirmConvertToInvoiceQst@1000 : TextConst 'ENU=Do you want to convert the quote to an invoice?;FRA=Souhaitez-vous transformer la demande de prix en factureÿ?';
      OpenNewInvoiceQst@1001 : TextConst 'ENU=The quote has been converted to invoice %1. Do you want to open the new invoice?;FRA=La demande de prix a ‚t‚ convertie en facture %1. Souhaitez-vous ouvrir la nouvelle factureÿ?';

    BEGIN
    END.
  }
}
