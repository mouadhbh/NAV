    LOCAL PROCEDURE CallUpdateCreditLimit@1000000000();
    VAR
      CreditLimitCaalculated@1000000000 : Decimal;
      CreditLimitActual@1000000001 : Decimal;
    BEGIN
      CreditLimitCaalculated:=Rec.CalculateCreditLimit;

      IF CreditLimitCaalculated=Rec."Credit Limit (LCY)" THEN BEGIN
        MESSAGE(Text90003);
        EXIT;
      END;

      IF GUIALLOWED AND NOT CONFIRM(
        Text90001,
        FALSE,
        FIELDCAPTION("Credit Limit (LCY)"),
        CreditLimitCaalculated)
        THEN
        EXIT;

      CreditLimitActual:=CreditLimitCaalculated;
      Rec.UpdateCreditLimit(CreditLimitActual);
      IF CreditLimitActual<>CreditLimitCaalculated THEN
        MESSAGE(Text90002,CreditLimitActual);
    END;

    BEGIN
    END.
